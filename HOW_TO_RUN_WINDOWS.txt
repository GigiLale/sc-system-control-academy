function loadState() {

function getTrialStart() {
  let started = localStorage.getItem(TRIAL_KEY);
  if (!started) {
    started = String(Date.now());
    localStorage.setItem(TRIAL_KEY, started);
  }
  return Number(started);
}

function trialInfo() {
  const days = Number(CONFIG.FREE_TRIAL_DAYS || 3);
  const started = getTrialStart();
  const expiresAt = started + days * 24 * 60 * 60 * 1000;
  const remaining = Math.max(0, expiresAt - Date.now());
  return {
    days,
    started,
    expiresAt,
    remaining,
    active: remaining > 0
  };
}

function trialLabel() {
  if (isPaid()) return 'Lifetime';
  const t = trialInfo();
  if (!t.active) return 'Trial expired';
  const hours = Math.ceil(t.remaining / (60 * 60 * 1000));
  if (hours <= 24) return `${hours}h trial left`;
  return `${Math.ceil(hours / 24)}d trial left`;
}

function hasTrialAccess() {
  return trialInfo().active;
}

function clamp(n, min, max) {

function exampleFor(skill, theme, level) {
  const complexity = difficultyLabel(level);
  if (skill === 'Python') return `# ${complexity} Python example\nitems = [3, 7, 2]\ntotal = sum(items)\nprint(total)`;
  if (skill === 'JavaScript') return `// ${complexity} JavaScript example\nconst items = [3, 7, 2];\nconst total = items.reduce((a, b) => a + b, 0);\nconsole.log(total);`;
  if (skill === 'C Programming') return `// ${complexity} C example\nint total = 0;\nfor (int i = 0; i < 3; i++) { total += numbers[i]; }`;
  if (skill === 'SQL') return `-- ${complexity} SQL example\nSELECT department, COUNT(*) AS total\nFROM employees\nGROUP BY department;`;
  if (skill === 'Power BI / DAX') return `-- ${complexity} DAX example\nTotal Sales = SUM(Sales[Amount])`;
  if (skill === 'HTML / CSS') return `<section class="card">\n  <h2>${theme}</h2>\n</section>`;
  if (skill === 'React') return `function Card({ title }) {\n  return <h2>{title}</h2>;\n}`;
  if (skill === 'Node.js / APIs') return `app.get('/api/status', (req, res) => {\n  res.json({ ok: true });\n});`;
  if (skill === 'Algorithms') return `function maxValue(items) {\n  let max = items[0];\n  for (const item of items) if (item > max) max = item;\n  return max;\n}`;
  return `Plan → Build → Test → Improve\nKeep code small, readable and safe.`;
}

function hashSeed(text) {
  let h = 2166136261;
  for (let i = 0; i < String(text).length; i++) {
    h ^= String(text).charCodeAt(i);
    h = Math.imul(h, 16777619);
  }
  return Math.abs(h >>> 0);
}

function seededRandom(seed) {
  let value = seed >>> 0;
  return function () {
    value = Math.imul(1664525, value) + 1013904223;
    return ((value >>> 0) / 4294967296);
  };
}

function shuffledOptions(question, level, index, retry) {
  const options = [question.correct, ...question.wrong];
  const seed = hashSeed(`${level}-${index}-${retry}-${question.question}-${question.correct}-${Date.now()}`);
  const random = seededRandom(seed);
  for (let i = options.length - 1; i > 0; i--) {
    const j = Math.floor(random() * (i + 1));
    [options[i], options[j]] = [options[j], options[i]];
  }

  // Force the correct answer into a different-looking slot per level/question.
  // This prevents the answer from repeatedly appearing in the same position.
  const targetIndex = (level + index + (retry ? 2 : 0)) % options.length;
  const currentIndex = options.indexOf(question.correct);
  if (currentIndex !== targetIndex) {
    [options[currentIndex], options[targetIndex]] = [options[targetIndex], options[currentIndex]];
  }
  return options;
}

      };
    case 27:
      return { ...common,
        question: `${retryTone}A JavaScript fetch request returns JSON. What is the safest next step before using it?`,
        correct: 'Check the response status and validate the data shape.',
        wrong: ['Render it immediately because JSON is always safe.', 'Convert it to CSS so the browser can trust it.', 'Ignore missing fields because APIs always return the same result.'],
        explain: 'APIs can fail or change. Strong apps check status, parse carefully and validate expected fields.'
      };
    case 28:
      return { ...common,
        question: `${retryTone}A C program writes outside the size of an array. What is the main risk?`,
        correct: 'Memory can be corrupted and the program may behave unpredictably.',
        wrong: ['The array becomes a database table automatically.', 'The compiler fixes the index at runtime every time.', 'The UI slows down but the data remains guaranteed safe.'],
        explain: 'C requires careful memory thinking. Out-of-bounds access is dangerous and can corrupt data.'
      };
    case 29:
      return { ...common,
        question: `${retryTone}At higher levels, why is random guessing a bad strategy in SC?`,
        correct: 'Wrong answers reduce XP and failed levels block progression until retried properly.',
        wrong: ['Guessing unlocks later content if done quickly enough.', 'The system rewards speed more than reasoning.', 'The correct answer is always the longest option.'],
        explain: 'The system is designed to reward understanding. Guessing can cost XP and delay progression.'
      };
    default:
      return { ...common,
        question: `${retryTone}At Level ${level}, what is the best sign you are ready for the next level?`,
        correct: 'You can explain the lesson, answer most questions correctly and apply the idea.',
        wrong: ['You recognise one repeated wording pattern from earlier levels.', 'You moved through the cards quickly and guessed confidently.', 'You remember the key phrase but cannot apply it in a new case.'],
        explain: 'Progress should require understanding, not luck. A 3/4 pass mark allows one mistake but still demands competence.'
      };
  }
}

function strengthenDistractors(q, level, index) {
  // Make distractors believable and avoid the common pattern where the longest answer is correct.
  // As levels rise, the options become closer in meaning and require more reasoning.
  const seriousReplacements = [
    'It may work in one demo but fail when the input changes.',
    'It improves the interface but does not fix the underlying logic.',
    'It handles the common case but ignores edge cases and error paths.',
    'It makes the code shorter but less clear to review or maintain.',
    'It delays validation until after the result has already been used.',
    'It solves part of the problem while leaving the risky part unchanged.'
  ];

  const sillyPatterns = [/computer eventually gets bored/i, /variable name sounds correct/i, /page design is finished/i, /logo colour/i, /CSS file stored the user password/i, /answer from where it appeared/i, /clicking/i];
  let wrong = q.wrong.map((item, i) => {
    if (sillyPatterns.some(rx => rx.test(item))) {
      return seriousReplacements[(level + index + i) % seriousReplacements.length];
    }
    return item;
  });

  // At higher levels, make answers more nuanced.
  if (level > 80) {
    wrong = wrong.map((item, i) => {
      const addon = [
        ' when the first version is small.',
        ' if no user has reported it yet.',
        ' when speed feels more important than reliability.',
        ' if the output looks correct in one example.'
      ][(level + i) % 4];
      return item.length < q.correct.length - 15 ? item + addon : item;
    });
  }

  // If the correct answer is much longer than every wrong option, add realistic detail to one or two wrong answers.
  const longestWrong = Math.max(...wrong.map(w => w.length));
  if (q.correct.length > longestWrong + 22) {
    wrong = wrong.map((item, i) => i < 2 ? item + ' while keeping the existing behaviour predictable.' : item);
  }

  // Sometimes shorten the correct wording without changing the idea.
  let correct = q.correct;
  const shortCorrect = [
    'Validate first and handle unexpected values.',
    'Use a reusable function or component.',
    'Trace values step by step.',
    'Keep it simple, readable and testable.',
    'Handle loading, errors and validation.'
  ];
  if (level > 40 && q.correct.length > 70 && (level + index) % 5 === 0) {
    correct = shortCorrect[(level + index) % shortCorrect.length];
  }

  return { ...q, correct, wrong };
}

function makeQuestion(level, index, retry = false) {
  let q = questionBlueprint(level, index, retry);
  q = strengthenDistractors(q, level, index);
  return { ...q, options: shuffledOptions(q, level, index, retry) };
}
function generateQuestions(level, retry = false) {
  const count = questionCount(level);
  const isCheckpoint = level % CONFIG.TEST_INTERVAL === 0;
  if (!isCheckpoint) {
    // Normal levels now ask four different questions about the same level topic.
    return Array.from({ length: count }, (_, i) => makeQuestion(level, i, retry));
  }

  // Checkpoints still refresh the previous 20 levels, but each question uses a different source level.
  const start = level - CONFIG.TEST_INTERVAL + 1;
  return Array.from({ length: count }, (_, i) => makeQuestion(start + (i % CONFIG.TEST_INTERVAL), i, retry));
}


function canAccess(level) {
  if (level <= CONFIG.FREE_LEVELS) return true;
  return isPaid() || hasTrialAccess();
}

function isUnlocked(level) {
  return level <= state.unlockedLevel;
}

function renderPath() {
  const path = document.getElementById('levelPath');
  path.innerHTML = '';
  const end = Math.min(visibleChapterStart + CONFIG.TEST_INTERVAL - 1, CONFIG.TOTAL_LEVELS);
  document.getElementById('chapterLabel').textContent = `Levels ${visibleChapterStart}–${end}`;
  document.getElementById('chapterTitle').textContent = `Levels ${visibleChapterStart}–${end}`;

  for (let level = visibleChapterStart; level <= end; level++) {
    const btn = document.createElement('button');
    btn.className = 'level-node';
    if (level === state.currentLevel) btn.classList.add('current');
    if (state.completedLevels.includes(level)) btn.classList.add('done');
    if (!isUnlocked(level) || !canAccess(level)) btn.classList.add('locked');
    if (level % CONFIG.TEST_INTERVAL === 0) btn.classList.add('test');

    const lockedLabel = !canAccess(level) ? 'Trial ended' : (!isUnlocked(level) ? 'Locked' : (level % CONFIG.TEST_INTERVAL === 0 ? 'Test' : difficultyLabel(level))); 
    btn.innerHTML = `
      <span class="level-number">${level}</span>
      <span class="level-title">${lockedLabel}</span>
    `;
    btn.addEventListener('click', () => openLevel(level));
    path.appendChild(btn);
  }
}

function openLevel(level) {
  activeLevel = level;
  if (!canAccess(level)) {
    showLessonView();
    renderPaywall(level);
    return;
  }
  if (!isUnlocked(level)) {
    showLessonView();
    renderLocked(level);
    return;
  }
  lessonStage = 'lesson';
  lessonCardIndex = 0;
  questionIndex = 0;
  selectedAnswer = null;
  currentQuestions = generateQuestions(level);
  testResults = [];
  showLessonView();
  renderLessonCard();
}

function renderQuestion() {
  const q = currentQuestions[questionIndex];
  const content = document.getElementById('lessonContent');
  const isTest = activeLevel % CONFIG.TEST_INTERVAL === 0;
  content.innerHTML = `
    <div class="question-card">
      <div class="question-meta">
        <span>${isTest ? 'Checkpoint test' : 'Level quiz'} • Question ${questionIndex + 1}/${currentQuestions.length}</span>
        <span>+${isTest ? 25 : 15} XP correct • -${CONFIG.WRONG_ANSWER_XP_PENALTY || 10} XP wrong</span>
      </div>
      <h2>${q.question}</h2>
      <div class="options" id="options"></div>
      <div id="feedbackBox"></div>
    </div>
  `;
  const options = document.getElementById('options');
  q.options.forEach((option) => {
    const btn = document.createElement('button');
    btn.className = 'option-btn';
    btn.textContent = option;
    btn.addEventListener('click', () => handleAnswer(option));
    options.appendChild(btn);
  });
}

function handleAnswer(answer) {
  if (selectedAnswer !== null) return;
  selectedAnswer = answer;
  const q = currentQuestions[questionIndex];
  const buttons = [...document.querySelectorAll('.option-btn')];
  const correct = answer === q.correct;

  buttons.forEach((btn) => {
    btn.disabled = true;
    if (btn.textContent === q.correct) btn.classList.add('correct');
    if (btn.textContent === answer && !correct) btn.classList.add('wrong');
  });

  testResults.push({ question: q.question, correct, topic: q.topic, chosen: answer, expected: q.correct, explain: q.explain });

  const feedback = document.getElementById('feedbackBox');
  if (correct) {
    const points = activeLevel % CONFIG.TEST_INTERVAL === 0 ? 25 : 15;
    state.xp += points;
    state.streak += 1;
    saveState();
    renderStats();
    feedback.innerHTML = `
      <div class="feedback pass">
        <strong>Correct.</strong> ${q.explain}
      </div>
      <div class="action-row">
        <button class="primary-btn" id="continueQuestion">${questionIndex === currentQuestions.length - 1 ? 'See result' : 'Next question'}</button>
      </div>
    `;
  } else {
    state.streak = 0;
    state.xp = Math.max(0, state.xp - (CONFIG.WRONG_ANSWER_XP_PENALTY || 10));
    state.failedTopics.push(q.topic);
    state.attempts[activeLevel] = (state.attempts[activeLevel] || 0) + 1;
    saveState();
    renderStats();
    feedback.innerHTML = `
      <div class="feedback fail">
        <strong>Not correct.</strong><br>
        You chose: <strong>${escapeHtml(answer)}</strong><br>
        Correct answer: <strong>${escapeHtml(q.correct)}</strong><br><br>
        Why: ${q.explain}<br><br>
        You lost ${CONFIG.WRONG_ANSWER_XP_PENALTY || 10} XP for rushing. Keep going. You need ${passNeeded()} correct answers to pass this level.
      </div>
      <div class="action-row">
        <button class="primary-btn" id="continueQuestion">${questionIndex === currentQuestions.length - 1 ? 'See result' : 'Next question'}</button>
      </div>
    `;
  }

  document.getElementById('continueQuestion').addEventListener('click', () => {
    if (questionIndex < currentQuestions.length - 1) {
      questionIndex++;
      selectedAnswer = null;
      renderQuestion();
    } else {
      finishAttempt();
    }
  });
}

function passNeeded() {
  return Math.ceil(currentQuestions.length * (CONFIG.PASS_MARK_PERCENT / 100));
}

function finishAttempt() {
  const correctCount = testResults.filter(r => r.correct).length;
  const total = currentQuestions.length;
  const percent = Math.round((correctCount / total) * 100);
  const needed = passNeeded();

  if (correctCount >= needed) {
    completeLevel(correctCount, total, percent);
  } else {
    renderRetryRecap(correctCount, total, percent);
  }
}

function renderRetryRecap(correctCount, total, percent) {
  const isTest = activeLevel % CONFIG.TEST_INTERVAL === 0;
  const wrong = testResults.filter(r => !r.correct);
  const rows = wrong.map((r, idx) => `
    <li>
      <strong>${idx + 1}. ${escapeHtml(r.topic)}</strong><br>
      Question: ${escapeHtml(r.question)}<br>
      Your answer: <span style="color:#991b1b">${escapeHtml(r.chosen)}</span><br>
      Correct idea: <span style="color:#166534">${escapeHtml(r.expected)}</span><br>
      Why: ${escapeHtml(r.explain)}
    </li>
  `).join('');

  const content = document.getElementById('lessonContent');
  content.innerHTML = `
    <div class="feedback fail">
      <h2>${isTest ? 'Checkpoint not passed yet' : 'Level not passed yet'}</h2>
      <p>You scored ${correctCount}/${total} (${percent}%). You need at least ${passNeeded()}/${total} correct to pass.</p>
      <p>Review the weak parts below. Then you will get new questions written differently, so you cannot just memorise the visible answer.</p>
      <ol>${rows}</ol>
    </div>
    <div class="action-row">
      <button class="primary-btn" id="retryWeakQuestions">Retry with new questions</button>
      <button class="secondary-btn" id="backToLesson">Review lesson first</button>
    </div>
  `;
  document.getElementById('retryWeakQuestions').addEventListener('click', () => retryQuiz(activeLevel));
  document.getElementById('backToLesson').addEventListener('click', () => restartLevel(activeLevel));
}

function retryQuiz(level) {
  activeLevel = level;
  lessonStage = 'quiz';
  questionIndex = 0;
  selectedAnswer = null;
  currentQuestions = generateQuestions(level, true);
  testResults = [];
  renderQuestion();
}

function restartLevel(level) {
  activeLevel = level;
  lessonStage = 'lesson';
  lessonCardIndex = 0;
  questionIndex = 0;
  selectedAnswer = null;
  currentQuestions = generateQuestions(level, false);
  testResults = [];
  renderLessonCard();
}

function completeLevel(correctCount, total, percent) {
  const isTest = activeLevel % CONFIG.TEST_INTERVAL === 0;

  if (!state.completedLevels.includes(activeLevel)) state.completedLevels.push(activeLevel);
  if (activeLevel >= state.unlockedLevel && activeLevel < CONFIG.TOTAL_LEVELS) {
    state.unlockedLevel = activeLevel + 1;
  }
  state.currentLevel = Math.min(activeLevel + 1, CONFIG.TOTAL_LEVELS);
  if (isTest && !state.testsPassed.includes(activeLevel)) state.testsPassed.push(activeLevel);
  saveState();
  visibleChapterStart = chapterStartFor(state.currentLevel);

  const content = document.getElementById('lessonContent');
  content.innerHTML = `
    <div class="feedback pass">
      <h2>${isTest ? 'Checkpoint passed' : 'Level passed'}</h2>
      <p>You scored ${correctCount}/${total} (${percent}%). Pass rule: 3/4 or better. Random clicking is risky because wrong answers reduce XP and failed levels do not unlock the next level.</p>
      <p>You unlocked Level ${state.currentLevel}. No certificate is issued. This platform focuses on learning progress, XP and checkpoint tests only.</p>
      <p><strong>XP:</strong> ${state.xp} • <strong>Streak:</strong> ${state.streak}</p>
    </div>
    <div class="action-row">
      <button class="primary-btn" id="goNextLevel">Continue to Level ${state.currentLevel}</button>
      <button class="secondary-btn" id="goPath">Back to path</button>
    </div>
  `;
  document.getElementById('goNextLevel').addEventListener('click', () => openLevel(state.currentLevel));
  document.getElementById('goPath').addEventListener('click', showPathView);
  renderAll();
}

function renderDashboard() {
  const completed = state.completedLevels.length;
  const tests = state.testsPassed.length;
  const weak = mostCommon(state.failedTopics).slice(0, 5);
  document.getElementById('progressDashboard').innerHTML = `
    <div class="metric-card"><strong>${completed}</strong><span>Levels completed</span></div>
    <div class="metric-card"><strong>${tests}</strong><span>Checkpoint tests passed</span></div>
    <div class="metric-card"><strong>${state.xp}</strong><span>Total XP earned</span></div>
    <div class="metric-card"><strong>${state.unlockedLevel}</strong><span>Highest unlocked level</span></div>
    <div class="metric-card" style="grid-column:1/-1">
      <h3>Weak areas to revise</h3>
      ${weak.length ? `<ul>${weak.map(x => `<li>${escapeHtml(x.name)} — missed ${x.count} time(s)</li>`).join('')}</ul>` : '<p>No weak areas recorded yet. Keep going.</p>'}
    </div>
  `;
}

function mostCommon(items) {
  const map = new Map();
  items.forEach(item => map.set(item, (map.get(item) || 0) + 1));
  return [...map.entries()].map(([name, count]) => ({ name, count })).sort((a, b) => b.count - a.count);
}

function setActiveNav(id) {
  document.querySelectorAll('.nav-btn').forEach(b => b.classList.remove('active'));

}

function escapeHtml(str) {
  return String(str).replace(/[&<>'"]/g, ch => ({ '&': '&amp;', '<': '&lt;', '>': '&gt;', "'": '&#39;', '"': '&quot;' }[ch]));
}
